58 static GDExtensionClassCallVirtual get_virtual_func(void *p_userdata, GDExtensionConstStringNamePtr p_name, uint32_t p_hash) {
59 // This is called by Godot the first time it calls a virtual function, and it caches the result, per object instance.
60 // Because of this, it can happen from different threads at once.
61 // It should be ok not using any mutex as long as we only READ data.
62 const StringName *class_name = reinterpret_cast<const StringName *>(p_userdata);
63 const StringName *name = reinterpret_cast<const StringName *>(p_name);
64
65 ERR_FAIL_COND_V_MSG(virtual_methods.find(*name) == virtual_methods.end(), nullptr, String("Virtual method '{0}' doesn't exist.").format(Array::make(*name)));
66
67 return virtual_methods[*name];
68 }
